# AgentFlow — v14 Cleanup
**Date**: 2026-04-17
**Цель**: убрать `__signature__` хак, довести покрытие v14 до плана
**Executor**: Codex
**Blocker**: `sdk/agentflow/async_client.py:279` и `sdk/agentflow/client.py` содержат `__init__.__signature__ = inspect.Signature(...)` — маскировка подписи от интроспекции

## Проблема

v14-hotfix закрыл backcompat, но через хак: реальный `__init__` принимает 6 параметров, а `inspect.signature()` возвращает 4 через переопределение `__signature__`. Это:
- Ломает auto-docs (Sphinx, pdoc) — показывают ложную сигнатуру
- Ломает IDE completion в сторонних проектах (зависит от инструмента)
- Запутывает при code review — сигнатура в коде ≠ сигнатура при runtime introspection
- Создаёт неочевидную связь между файлом класса и backcompat-тестом

## Решение

Убрать `retry_policy` и `circuit_breaker` из `__init__`, вынести в метод `configure_resilience(...)` (как в оригинальном v14 плане). Подпись `__init__` честная, runtime поведение то же.

### `tests/sdk/test_retry.py` — минимум 6 тестов

```python
import time
import pytest
from sdk.agentflow.retry import RetryPolicy, is_retryable_method, RETRYABLE_STATUS


def test_retry_policy_exponential_backoff():
    p = RetryPolicy(max_attempts=5, initial_delay_s=0.1, jitter_factor=0)
    assert p.compute_delay(0) == pytest.approx(0.1)
    assert p.compute_delay(1) == pytest.approx(0.2)
    assert p.compute_delay(2) == pytest.approx(0.4)
    assert p.compute_delay(3) == pytest.approx(0.8)


def test_retry_policy_respects_retry_after():
    p = RetryPolicy(initial_delay_s=0.1)
    assert p.compute_delay(0, retry_after_s=3.0) == 3.0
    # retry_after превышает max_delay → clamped
    assert p.compute_delay(0, retry_after_s=999.0) <= p.max_delay_s


def test_retry_policy_caps_at_max_delay():
    p = RetryPolicy(max_delay_s=1.0, jitter_factor=0)
    assert p.compute_delay(10) == 1.0


def test_retry_policy_jitter_stays_in_bounds():
    p = RetryPolicy(initial_delay_s=1.0, jitter_factor=0.5)
    samples = [p.compute_delay(0) for _ in range(100)]
    assert all(0.5 <= s <= 1.5 for s in samples)


def test_is_retryable_method_only_idempotent():
    assert is_retryable_method("GET") is True
    assert is_retryable_method("HEAD") is True
    assert is_retryable_method("PUT") is True
    assert is_retryable_method("DELETE") is True
    assert is_retryable_method("POST") is False


def test_retryable_statuses():
    assert 429 in RETRYABLE_STATUS
    assert 503 in RETRYABLE_STATUS
    assert 504 in RETRYABLE_STATUS
    assert 200 not in RETRYABLE_STATUS
    assert 404 not in RETRYABLE_STATUS
```

### `tests/sdk/test_circuit_breaker.py` — минимум 6 тестов

```python
import time
import pytest
from sdk.agentflow.circuit_breaker import CircuitBreaker, CircuitState, CircuitOpenError


def test_breaker_starts_closed():
    b = CircuitBreaker()
    assert b.state == CircuitState.CLOSED
    b.before_call()  # не должно бросать


def test_breaker_opens_after_threshold():
    b = CircuitBreaker(failure_threshold=3)
    for _ in range(3):
        b.record_failure()
    assert b.state == CircuitState.OPEN
    with pytest.raises(CircuitOpenError):
        b.before_call()


def test_breaker_resets_to_half_open_after_timeout():
    b = CircuitBreaker(failure_threshold=1, reset_timeout_s=0.1)
    b.record_failure()
    assert b.state == CircuitState.OPEN
    time.sleep(0.15)
    b.before_call()
    assert b.state == CircuitState.HALF_OPEN


def test_breaker_half_open_success_closes():
    b = CircuitBreaker(failure_threshold=1, reset_timeout_s=0.1)
    b.record_failure()
    time.sleep(0.15)
    b.before_call()
    b.record_success()
    assert b.state == CircuitState.CLOSED


def test_breaker_half_open_failure_reopens():
    b = CircuitBreaker(failure_threshold=1, reset_timeout_s=0.1)
    b.record_failure()
    time.sleep(0.15)
    b.before_call()
    b.record_failure()
    assert b.state == CircuitState.OPEN


def test_breaker_half_open_allows_one_probe():
    b = CircuitBreaker(failure_threshold=1, reset_timeout_s=0.1, half_open_max_calls=1)
    b.record_failure()
    time.sleep(0.15)
    b.before_call()  # probe 1 — ok
    with pytest.raises(CircuitOpenError):
        b.before_call()  # probe 2 — заблокирован


def test_breaker_success_resets_failure_counter():
    b = CircuitBreaker(failure_threshold=3)
    b.record_failure()
    b.record_failure()
    b.record_success()
    b.record_failure()
    b.record_failure()
    # 2 + (reset) + 2 = 2, меньше порога — закрыт
    assert b.state == CircuitState.CLOSED
```
