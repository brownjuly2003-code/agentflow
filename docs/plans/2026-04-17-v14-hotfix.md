## TASK 1 — Fix backwards compat (Python SDK)

**Подход:** retry/circuit-breaker НЕ через kwargs конструктора — через публичные атрибуты/setter. Это сохраняет точную подпись `__init__`, и пользователи могут включить resilience опционально после создания клиента.

### Изменения в `sdk/agentflow/async_client.py`

```python
class AsyncAgentFlowClient:
    def __init__(
        self,
        base_url: str,
        api_key: str,
        timeout: float = 10.0,
    ):
        self._http = httpx.AsyncClient(
            base_url=base_url.rstrip("/"),
            timeout=timeout,
            headers={"X-API-Key": api_key},
        )
        self.retry_policy: RetryPolicy = RetryPolicy()
        self.circuit_breaker: CircuitBreaker = CircuitBreaker()

    def configure_resilience(
        self,
        retry_policy: RetryPolicy | None = None,
        circuit_breaker: CircuitBreaker | None = None,
    ) -> "AsyncAgentFlowClient":
        """Optional resilience configuration. Returns self for chaining."""
        if retry_policy is not None:
            self.retry_policy = retry_policy
        if circuit_breaker is not None:
            self.circuit_breaker = circuit_breaker
        return self
```

**Аналогично** в `sdk/agentflow/client.py`.

### Использование в хелперах `_request`

```python
async def _request(self, method, path, ...):
    self.circuit_breaker.before_call()
    # attempt loop с self.retry_policy
    ...
```

### Пример в docstring / README SDK

```python
# Default (sane defaults for resilience):
client = AsyncAgentFlowClient("https://api.example.com", "key")

# Custom:
client = AsyncAgentFlowClient("https://api.example.com", "key")
client.configure_resilience(
    retry_policy=RetryPolicy(max_attempts=5, initial_delay_s=0.5),
    circuit_breaker=CircuitBreaker(failure_threshold=10),
)
```

class AgentFlowClient {
    public retryPolicy: RetryPolicy;
    public circuitBreaker: CircuitBreaker;

    constructor(baseUrl: string, apiKey: string, timeout: number = 10000) {
        // existing code...
        this.retryPolicy = new RetryPolicy();
        this.circuitBreaker = new CircuitBreaker();
    }

    configureResilience(options: {
        retryPolicy?: RetryPolicy;
        circuitBreaker?: CircuitBreaker;
    }): this {
        if (options.retryPolicy) this.retryPolicy = options.retryPolicy;
        if (options.circuitBreaker) this.circuitBreaker = options.circuitBreaker;
        return this;
    }
}
