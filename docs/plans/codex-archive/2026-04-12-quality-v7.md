```python
"""
Локальный запуск всех security проверок.
Используй перед PR: python scripts/security_check.py

Запускает:
  1. bandit -r src/ sdk/
  2. safety check
  3. pip-audit (альтернатива safety, более свежая)
  4. Выводит сводный отчёт
"""
import subprocess, sys

checks = [
    ("Bandit (SAST)", ["bandit", "-r", "src/", "sdk/", "-c", ".bandit", "--severity-level", "medium"]),
    ("Safety (CVE)", ["safety", "check", "-r", "requirements.txt"]),
    ("pip-audit",    ["pip-audit", "--requirement", "requirements.txt"]),
]

failed = []
for name, cmd in checks:
    print(f"\n{'='*50}\n{name}\n{'='*50}")
    result = subprocess.run(cmd)
    if result.returncode != 0:
        failed.append(name)

if failed:
    print(f"\n❌ Failed: {', '.join(failed)}")
    sys.exit(1)
else:
    print("\n✅ All security checks passed")
```

```python
    "POST /v1/batch":             (200, 0.02),
    "GET /v1/health":             (20,  0.00),   # health должен быть всегда быстрым

    "users": 50,
    "spawn_rate": 10,
```

### run_load_test.py

```python
"""
Запускает Locust и проверяет результаты против thresholds.

Usage:
  python tests/load/run_load_test.py --host http://localhost:8000

Exit code:
  0 — все thresholds выполнены
  1 — есть нарушения (CI падает)
"""
import json, subprocess, sys
from pathlib import Path
from tests.load.thresholds import THRESHOLDS, LOAD_PROFILE

def run_locust(host: str) -> dict:
    result_file = Path("tests/load/results.json")
    subprocess.run([
        "locust", "-f", "tests/load/locustfile.py",
        "--host", host,
        "--users", str(LOAD_PROFILE["users"]),
        "--spawn-rate", str(LOAD_PROFILE["spawn_rate"]),
        "--run-time", LOAD_PROFILE["run_time"],
        "--headless",
        "--json", str(result_file),
        "--exit-code-on-error", "0",  # не падать на ошибках запросов, мы сами проверим
    ], check=True)
    return json.loads(result_file.read_text())

def check_thresholds(stats: dict) -> list[str]:
    violations = []
    for entry in stats:
        name = entry["name"]
        for pattern, (p95_max, error_max) in THRESHOLDS.items():
            if pattern.split("{")[0].rstrip("/") in name:
                p95 = entry.get("response_time_percentile_0.95", 0)
                error_rate = entry.get("fail_ratio", 0)
                if p95 > p95_max:
                    violations.append(f"{name}: p95={p95}ms > threshold {p95_max}ms")
                if error_rate > error_max:
                    violations.append(f"{name}: error_rate={error_rate:.1%} > threshold {error_max:.1%}")
    return violations

if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--host", default="http://localhost:8000")
    args = parser.parse_args()

    stats = run_locust(args.host)
    violations = check_thresholds(stats)

    if violations:
        print("\n❌ Performance threshold violations:")
        for v in violations:
            print(f"  • {v}")
        sys.exit(1)
    else:
        print("\n✅ All performance thresholds met")
```

### session_aggregation.py

```python
"""
Stateful Flink operator: агрегация пользовательских сессий.

Логика:
- Сессия = последовательность событий одного user с gap < 30 минут
- Состояние: last_event_time per user (MapState)
- Output: session_start, session_end, event_count, total_value

Это демонстрирует:
1. Stateful processing (MapState)
2. Session windows (gap-based)
3. Checkpointing (состояние переживает рестарт)
"""
from pyflink.datastream import StreamExecutionEnvironment
from pyflink.datastream.state import MapStateDescriptor
from pyflink.datastream.functions import KeyedProcessFunction
from pyflink.common.typeinfo import Types
from pyflink.common.watermark_strategy import WatermarkStrategy
import json
from datetime import datetime, timedelta

SESSION_GAP = timedelta(minutes=30)

class SessionAggregator(KeyedProcessFunction):
    """
    Keyed по user_id. Отслеживает сессии через MapState.
    При gap > 30 мин — закрывает сессию и эмитит результат.
    """

    def open(self, runtime_context):
        # MapState: user_id → session_state (start_time, last_time, event_count, total_value)
        descriptor = MapStateDescriptor(
            "session_state",
            Types.STRING(),
            Types.PICKLED_BYTE_ARRAY(),
        )
        self.state = runtime_context.get_map_state(descriptor)

    def process_element(self, event: dict, ctx: KeyedProcessFunction.Context):
        user_id = event["user_id"]
        event_time = datetime.fromisoformat(event["timestamp"])
        value = float(event.get("value", 0))

        if self.state.contains(user_id):
            session = self.state.get(user_id)
            last_time = datetime.fromisoformat(session["last_time"])

            if event_time - last_time > SESSION_GAP:
                # Закрываем старую сессию
                yield {
                    "user_id": user_id,
                    "session_start": session["start_time"],
                    "session_end": session["last_time"],
                    "event_count": session["event_count"],
                    "total_value": session["total_value"],
                    "status": "closed",
                }
                # Начинаем новую
                self.state.put(user_id, {
                    "start_time": event_time.isoformat(),
                    "last_time": event_time.isoformat(),
                    "event_count": 1,
                    "total_value": value,
                })
            else:
                # Обновляем текущую сессию
                session["last_time"] = event_time.isoformat()
                session["event_count"] += 1
                session["total_value"] += value
                self.state.put(user_id, session)
        else:
            # Новая сессия
            self.state.put(user_id, {
                "start_time": event_time.isoformat(),
                "last_time": event_time.isoformat(),
                "event_count": 1,
                "total_value": value,
            })

def build_session_pipeline(env: StreamExecutionEnvironment, source_topic: str, sink_topic: str):
    """Строит pipeline: Kafka source → SessionAggregator → Kafka sink."""
    from pyflink.datastream.connectors.kafka import KafkaSource, KafkaSink

    source = (KafkaSource.builder()
        .set_bootstrap_servers("kafka:9092")
        .set_topics(source_topic)
        .set_group_id("session-aggregator")
        .set_starting_offsets(...)
        .set_value_only_deserializer(...)
        .build())

    stream = (env
        .from_source(source, WatermarkStrategy.for_monotonous_timestamps(), "Kafka Source")
        .map(lambda raw: json.loads(raw))
        .key_by(lambda e: e["user_id"])
        .process(SessionAggregator()))

    return stream
```

```python
class ServingBackend(ABC):
    """
    Абстракция над serving store.
    Реализации: DuckDB (single-node), ClickHouse (distributed).
    """

    def execute(self, sql: str, params: list | None = None) -> list[dict]:

    @abstractmethod
    def get_entity(self, entity_type: str, entity_id: str, tenant_id: str) -> dict | None:
```

```python
# src/serving/backends/clickhouse_backend.py
import clickhouse_connect
from src.serving.backends import ServingBackend

class ClickHouseBackend(ServingBackend):
    """
    ClickHouse backend для production масштаба.

    Преимущества перед DuckDB:
    - Горизонтальное масштабирование
    - Встроенная репликация
    - Columnar хранилище оптимизировано под аналитику
    - Сжатие данных out-of-the-box

    Конфигурация через env vars:
    - CLICKHOUSE_HOST (default: localhost)
    - CLICKHOUSE_PORT (default: 8123)
    - CLICKHOUSE_USER (default: default)
    - CLICKHOUSE_PASSWORD
    - CLICKHOUSE_DATABASE (default: agentflow)
    """
    def __init__(self, host: str, port: int, user: str, password: str, database: str):
        self._client = clickhouse_connect.get_client(
            host=host, port=port, user=user, password=password, database=database
        )
        self._database = database

    def execute(self, sql: str, params: list | None = None) -> list[dict]:
        result = self._client.query(sql, parameters=params or [])
        return [dict(zip(result.column_names, row)) for row in result.result_rows]

    def get_entity(self, entity_type: str, entity_id: str, tenant_id: str) -> dict | None:
        sql = f"""
            SELECT * FROM {self._database}.{entity_type}s
            WHERE {entity_type}_id = {{entity_id:String}}
              AND tenant_id = {{tenant_id:String}}
            LIMIT 1
        """
        rows = self._client.query(sql, parameters={"entity_id": entity_id, "tenant_id": tenant_id})
        if not rows.result_rows:
            return None
        return dict(zip(rows.column_names, rows.result_rows[0]))

    def health(self) -> dict:
        try:
            self._client.ping()
            return {"backend": "clickhouse", "status": "ok"}
        except Exception as e:
            return {"backend": "clickhouse", "status": "error", "error": str(e)}
```

### benchmark_compare.py

```python
"""
Сравнивает текущие результаты load test с baseline.
Регрессия: p95 вырос > 20% — CI падает.
Улучшение: p95 упал > 10% — обновляем baseline.

Usage:
  python scripts/benchmark_compare.py \
    --current tests/load/results.json \
    --baseline docs/benchmarks/baseline.json
"""
import json, sys
from pathlib import Path

REGRESSION_THRESHOLD = 0.20   # 20% деградация → блокер
IMPROVEMENT_THRESHOLD = 0.10  # 10% улучшение → обновляем baseline

def compare(current: dict, baseline: dict) -> tuple[list[str], list[str]]:
    regressions, improvements = [], []
    for endpoint, base_stats in baseline["endpoints"].items():
        if endpoint not in current:
            continue
        curr_p95 = current[endpoint]["p95"]
        base_p95 = base_stats["p95"]
        delta = (curr_p95 - base_p95) / base_p95
        if delta > REGRESSION_THRESHOLD:
            regressions.append(
                f"{endpoint}: p95 {base_p95}ms → {curr_p95}ms (+{delta:.1%}) REGRESSION"
            )
        elif delta < -IMPROVEMENT_THRESHOLD:
            improvements.append(
                f"{endpoint}: p95 {base_p95}ms → {curr_p95}ms ({delta:.1%}) IMPROVEMENT"
            )
    return regressions, improvements

if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument("--current", required=True)
    parser.add_argument("--baseline", required=True)
    parser.add_argument("--update-baseline", action="store_true")
    args = parser.parse_args()

    current = json.loads(Path(args.current).read_text())
    baseline = json.loads(Path(args.baseline).read_text())
    regressions, improvements = compare(current, baseline)

    for r in regressions: print(f"❌ {r}")
    for i in improvements: print(f"✅ {i}")

    if regressions:
        print(f"\n{len(regressions)} regression(s) found. PR blocked.")
        sys.exit(1)
    if improvements and args.update_baseline:
        # Обновить baseline
        ...
        print("Baseline updated.")
```
